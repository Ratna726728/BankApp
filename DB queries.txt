create database bank_account_user_details;

use bank_account_user_details;

create table user_registation (
	id int auto_increment primary key,
    firstName varchar(50) not null,
    surname varchar(50) not null,
    age int,
    email varchar(50),
    contactNumber varchar(50) not null,
    accountNumber varchar(10) not null,
    balance double,
    address varchar(50),
    userId varchar(20) not null,
    branchName varchar(50) not null
);

ALTER TABLE user_registation ADD UNIQUE (accountNumber);

ALTER TABLE user_registation ADD transactionType VARCHAR(10);

CREATE TABLE transaction_details (
    transactionId INT AUTO_INCREMENT PRIMARY KEY,
    accountNumber VARCHAR(10) NOT NULL,
    userId VARCHAR(20) NOT NULL,
    amount DOUBLE NOT NULL,
    transactionType VARCHAR(20) NOT NULL,
    dateOfTransaction TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (accountNumber)
    REFERENCES user_registation(accountNumber)
);
